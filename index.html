curry in js

function curry(fn) {
  var arity = fn.length;
  var args = [];
  function curried() {
    args = args.concat(
        Array.prototype.slice.call(arguments));
    if (arity <= args.length) {
      return fn.apply(null, args);
    }
    return curried;
  }
  return curried;
}
